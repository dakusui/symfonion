`mmml` is implemented as a wrapper for `buildweaver` utility.
It just generates

[[CONTENT]]
.mmml
----
#!/bin/env bash
set -E -o nounset -o errexit +o posix -o pipefail
shopt -s inherit_errexit

function main() {
  local _dir="${1}"
  shift
  buildweaver "${HOME}/.local/lib/mmml/latest/lib/json++/buildweaver.json" "${@}"
}

main "$(dirname ${0})" ${@}"
----
