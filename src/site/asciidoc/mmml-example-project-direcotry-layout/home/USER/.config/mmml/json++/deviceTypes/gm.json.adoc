[%nowrap, script]
.deviceType-helper.sh
----
function wave() {
  local _device_name="${1}" _wave_name="${2}"
  ref .system.devices."${_device_name}".waves."${_wave_name}"
}

function device_name() {
  local _p
  _p="$(cur)"
  echo "${_p##*.}"
}
----
